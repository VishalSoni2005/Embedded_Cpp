void main(void) {
  int i, j, k;

  TRISC = 0;
  LATC = 0;

  i = 0x02;
  j = 0x05;
  k = i + j;

  PORTA = i;
  PORTB = j;
  PORTC = k;
  return;
}

2: array Addition

#include <xc.h>
using namespace std;

void main( void ) {
  int sum = 0x00;

  int arr[4] = {0x01, 0x02, 0x03, 0x04};

  TRISC = 0;
  LATC = 0;

  for(int i=0; i<4; i++) {
    sum += arr[i];
  }

  PORTC = sum;

}
